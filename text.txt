apiVersion: v1
kind: Pod
metadata:
    name: multi-pv-pod
spec:
    containers:
    - name: my-container
        image: nginx
        volumeMounts:
        - mountPath: "/mnt/volume1"
            name: volume1
        - mountPath: "/mnt/volume2"
            name: volume2
    volumes:
    - name: volume1
        persistentVolumeClaim:
            claimName: pvc-volume1
    - name: volume2
        persistentVolumeClaim:
            claimName: pvc-volume2
